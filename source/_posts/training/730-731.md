## [225. 用队列实现栈](https://leetcode.cn/problems/implement-stack-using-queues/)

**解法：模拟**

~~~
class MyStack {
    ArrayDeque<Integer> q1 = new ArrayDeque<>();
    ArrayDeque<Integer> q2 = new ArrayDeque<>();
    public MyStack() {

    }
    
    public void push(int x) {
        q2.add(x);
        while (!q1.isEmpty()) q2.add(q1.poll());
        ArrayDeque<Integer> t = q1;
        q1 = q2;
        q2 = t;
    }
    
    public int pop() {
        return q1.poll();
    }
    
    public int top() {
        return q1.peek();
    }
    
    public boolean empty() {
        return q1.isEmpty();
    }
}

## [232. 用栈实现队列](https://leetcode.cn/problems/implement-queue-using-stacks/)

**解法：模拟**

~~~
class MyQueue {
    ArrayDeque<Integer> a, b;
    public MyQueue() {
        a = new ArrayDeque<>();
        b = new ArrayDeque<>();
    }
    
    public void push(int x) {
        while (!b.isEmpty()) a.push(b.pop());
        a.push(x);
    }
    
    public int pop() {
        int x = peek();
        b.pop();
        return x;
    }
    
    public int peek() {
        while (!a.isEmpty()) b.push(a.pop());
        return b.peek();
    }
    
    public boolean empty() {
        return a.isEmpty() && b.isEmpty();
    }
}

 * Your MyQueue object will be instantiated and called as such:
 * MyQueue obj = new MyQueue();
 * obj.push(x);
 * int param_2 = obj.pop();
 * int param_3 = obj.peek();
 * boolean param_4 = obj.empty();
 */
~~~

## [20. 有效的括号](https://leetcode.cn/problems/valid-parentheses/)

**解法：模拟**

~~~
class Solution {
    public boolean isValid(String s) {
        ArrayDeque<Character> stack = new ArrayDeque<>();
        for (char x : s.toCharArray()) {
            if (x == '(') stack.push(')');
            else if (x == '[') stack.push(']');
            else if (x == '{') stack.push('}');
            else {
                if (stack.size() > 0 && stack.peek() == x) stack.pop();
                else return false;
            }
        }
        return stack.isEmpty();
    }
}
~~~

## [622. 设计循环队列](https://leetcode.cn/problems/design-circular-queue/)

**解法：模拟**

~~~
class MyCircularQueue {
    int[] q;
    int hh = 0, tt = 0;
    public MyCircularQueue(int k) {
        //将长度开到k + 1，目的是区分队空和队满的情况
        q = new int[k + 1];
    }
    
    public boolean enQueue(int x) {
        if (isFull()) return false;
        q[tt ++ ] = x;
        if (tt == q.length) tt = 0;
        return true;
    }
    
    public boolean deQueue() {
        if (isEmpty()) return false;
        hh ++ ;
        if (hh == q.length) hh = 0;
        return true;
    }
    
    public int Front() {
        if (isEmpty()) return -1;
        return q[hh];
    }
    
    public int Rear() {
        if (isEmpty()) return -1;
        int t = tt - 1;
        return t < 0 ? q[q.length - 1] : q[t];
    }
    
    public boolean isEmpty() {
        return hh == tt;
    }
    
    public boolean isFull() {
        return (tt + 1) % q.length == hh;
    }
}
~~~

class Solution {
    public int minimumOperations(int[] nums) {
        Priority<Integer> q = new PriorityQueue<>();
        for (int x : nums) if (x > 0) q.add(x);
        int res = 0;
        while (!q.isEmpty()) {
            int t = q.poll();
            int num = q.size();
            while (num -- > 0) {
                int p = q.poll();
                res ++ ;
                p -= t;
                if (p > 0) q.add(p);
            }
        }
        return res;
    }
}
